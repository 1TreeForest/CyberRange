<!-- Web.Config 配置文件 -->



<configuration>

    <system.web>

        <customErrors mode="Off"/>

    </system.web>

</configuration>

<!-- Web.Config 配置文件 -->



<configuration>

    <system.web>

        <customErrors mode="RemoteOnly" defaultRedirect="mycustompage.htm"/>

    </system.web>

</configuration>
